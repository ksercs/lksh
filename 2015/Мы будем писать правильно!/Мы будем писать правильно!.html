Рекомендуемые правила форматирования кода:

  • Отступ два или четыре пробела; табуляции запрещены;
  • Вокруг знаков арифметических операций ставятся пробелы;
  • Вокруг присваиваний ставятся пробелы;
  • Процедуры, функции и другие смысловые блоки разделяются пустыми строками;
  • После запятой и двоеточия ставится пробел;
  • Между именем функции и скобками пробел не ставится, с внутренней стороны скобок пробел не ставится.
  • Ширина текста программы (то есть количество символов в одной строке кода) не должна превышать 80-100 символов;
  • В операторах if, while, do-while, for тело цикла заключается в фигурные скобки (составной оператор), даже если тело — единственный оператор;
  • Каждый новый оператор должен находиться в отдельной строке.
  • Идентификаторы должны иметь длинные (но в меру), осмысленные названия. Исключения допускаются только для счетчиков в циклах (i, j), а также для размеров исходных данных (N, M).
  • В идентификаторах используются только английские слова, транслитерация не допускается.

program sum;

function sum(a, b: longint): longint;
begin
  sum := a + b;
end;

var
  a, b, s: longint;

begin
  assign(input, 'aplusb.in');
  reset(input);
  assign(output, 'aplusb.out');
  rewrite(output);

  read(a, b);

  s := 0;
  if (a < 0) then begin
    a := -a;
    if (b < 0) then begin
      b := -b;
      s := a + b;
    end else begin
      while (s <= 0) do begin
        case a of
          1: begin
            s := s + 3;
          end;
          2: begin
            s := s - 4;
            a := a - 1;
          end;
          else
            s := 1;
        end;
      end;
    end;
  end else if (b < 0) then begin
    b := -b;
    s := (a + b) * (a - b);
  end else begin
    s := sum(a, b) * sum(a, b);
  end;
   
  writeln(s);

  close(input);
  close(output);
end.
 
#include <stdio.h>

int sum(int a, int b) {
  return a + b;

  int a;
  int b;
  int s;

  freopen("aplusb.in", "r", stdin);

  scanf("%d%d", &a, &b);

  s = 0;
  if (a < 0) {
    a = -a;
    if (b < 0) {
      b = -b;
      s = a + b;

          a = a - 1;
          break;
        default:
          s = 1;
        }
      }
    }
  } else if (b < 0) {
    b = -b;
    s = (a + b) * (a - b);
  } else {
    s = sum(a, b) * sum(a, b);
